
## Classi (OOP)

```python
class Persona:
    # Attributo di classe
    specie = "Homo sapiens"

    def __init__(self, nome: str, eta: int):
        self.nome = nome    # attributo di istanza
        self.eta = eta

    def saluta(self) -> str:
        return f"Ciao, sono {self.nome}"

    def __str__(self) -> str:
        return f"Persona({self.nome}, {self.eta})"

    def __repr__(self) -> str:
        return self.__str__()

# Ereditarietà
class Studente(Persona):
    def __init__(self, nome, eta, corso):
        super().__init__(nome, eta)
        self.corso = corso

    def saluta(self) -> str:
        return f"{super().saluta()}, studio {self.corso}"
```

## Gestione eccezioni

```python
try:
    risultato = 10 / 0
except ZeroDivisionError as e:
    print(f"Errore: {e}")
except (TypeError, ValueError) as e:
    print(f"Tipo o valore errato: {e}")
except Exception as e:
    print(f"Errore generico: {e}")
else:
    print("Nessun errore")
finally:
    print("Eseguito sempre")

# Lanciare eccezioni
raise ValueError("Valore non valido")
```

## Decoratori

```python
def logger(func):
    def wrapper(*args, **kwargs):
        print(f"Chiamata a {func.__name__}")
        return func(*args, **kwargs)
    return wrapper

@logger
def saluta(nome):
    return f"Ciao {nome}"
```

## Context Manager

```python
class Connessione:
    def __enter__(self):
        print("Connessione aperta")
        return self

    def __exit__(self, exc_type, exc_val, exc_tb):
        print("Connessione chiusa")

with Connessione() as c:
    print("Uso la connessione")
```

## Async/Await

```python
import asyncio

async def fetch_data():
    await asyncio.sleep(1)
    return "dati"

async def main():
    risultato = await fetch_data()
    print(risultato)

asyncio.run(main())
```
